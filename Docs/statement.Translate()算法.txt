///statement.Translate()算法///

if statement is IfStatement
	if statement2 == null
		L = newLabel()
		condition = expression.GetValue()
		(je, condition, 0, L)
		statement1.Translate()
		(label, _, _, L)
	else
		L1 = newLabel()
		L2 = newLabel()
		condition = expression.GetValue();
		(je, condition, 0, L1)
		statement1.Translate()
		(jmp, _, _, L2)
		(label, _, _, L1)
		statement2.Translate()
		(label, _, _, L2)

else if statement is WhileStatement
	L1 = newLabel()
	L2 = newLabel()
	(label, _, _, L1)
	condition = expression.GetValue()
	(je, condition, 0, L2)
	statement.Translate()
	(jmp, _, _, L1)
	(label, _, _, L2)

else if statement is DoWhileStatement
	L = newLabel()
	(label, _, _, L)
	statement.Translate()
	condition = expression.GetValue()
	(jne, condition, 0, L)

else if statement is ForStatement

else if statement is FieldDefineStatement
	newVar(name, type, 0)


else if statement is arrayDefineStatement

else if statement is AssignStatement

	varTable.SetValue(identify, expression.GetValue())
